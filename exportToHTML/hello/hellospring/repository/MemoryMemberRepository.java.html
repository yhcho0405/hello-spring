1    package hello.hellospring.repository;
2    
3    import hello.hellospring.domain.Member;
4    import org.springframework.stereotype.Repository;
5    
6    import java.util.*;
7    
8    public class MemoryMemberRepository implements MemberRepository {
9    
10       private static Map<Long, Member> store = new HashMap<>();
11       private static long sequence = 0L;
12   
13       @Override
14       public Member save(Member member) {
15           member.setId(++sequence);
16           store.put(member.getId(), member);
17           return member;
18       }
19   
20       @Override
21       public Optional<Member> findById(Long id) {
22           return Optional.ofNullable(store.get(id));
23       }
24   
25       @Override
26       public Optional<Member> findByName(String name) {
27           return store.values().stream()
28                   .filter(member -> member.getName().equals(name))
29                   .findAny();
30       }
31   
32       @Override
33       public List<Member> findAll() {
34           return new ArrayList<>(store.values());
35       }
36   
37       public void clearStore() {
38           store.clear();
39       }
40   }
