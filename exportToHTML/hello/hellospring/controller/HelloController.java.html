HelloController.java 
1    package hello.hellospring.controller;
2    
3    import org.springframework.stereotype.Controller;
4    import org.springframework.ui.Model;
5    import org.springframework.web.bind.annotation.GetMapping;
6    import org.springframework.web.bind.annotation.RequestParam;
7    import org.springframework.web.bind.annotation.ResponseBody;
8    
9    @Controller
10   public class HelloController {
11   
12       @GetMapping("hello") // localhost:8080/hello로 들어오면
13       public String hello(Model model) {
14           model.addAttribute("data", "Hello!!"); // model data에 "hello!!" 담아서
15           return "hello"; // templates/hello.html을 열어라
16       }
17   
18       @GetMapping("hello-mvc")
19       // @RequestParam("name") String name 때문에 localhost:8080/hello-mvc?name= 이런식으로
20       public String helloMvc(@RequestParam("name") String name, Model model) {
21           model.addAttribute("name", name);
22           return "hello-template";
23       }
24   
25   
26       // API 방식, 얜 veiw 같은거 없이 문자 그대로 보냄, 위에 템플릿 엔진을 사용한 방식들은 화면을 렌더링해서 보내줌
27       @GetMapping("hello-string")
28       @ResponseBody // http res body에 return을 직접 넣겠다.
29       public String helloString(@RequestParam("name") String name) {
30           return "hello " + name;
31       }
32   
33       // API방식 json으로 반환
34       @GetMapping("hello-api")
35       @ResponseBody
36       public Hello helloApi(@RequestParam("name") String name) {
37           Hello hello = new Hello();
38           hello.setName(name);
39           return hello;
40       }
41   
42       static class Hello {
43           private  String name;
44   
45           public String getName() {
46               return name;
47           }
48           public void setName(String name) {
49               this.name = name;
50           }
51       }
52   }
